\section{Hypothesis and Landscape Diagnostics}

\begin{frame}{Key Hypothesis: Close Substitute Solutions}
  \small
  \centering
  \begin{block}{Hypothesis}
    \centering
    Kidney exchange often contains many \textbf{close substitute exchange plans}.
  \end{block}

  \vspace{0.4em}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.50\textwidth}
      \textbf{Why KEP can behave this way}
      \begin{itemize}\tightitems
        \item A feasible plan is a packing of cycles and chains.
        \item One exchange can often be replaced by another.
        \item The plan identity changes, but its objective value may stay close.
      \end{itemize}

      \vspace{0.35em}
      \centering
      \begin{tikzpicture}[scale=0.82, every node/.style={transform shape}]
        \node[ksmall,draw=kepgreen!70,fill=kepgreen!7] (o1) at (0,0) {Oracle plan};
        \node[ksmall,below=8mm of o1,draw=keporange!75,fill=keporange!8] (t1) {Two-stage plan};
        \node[font=\scriptsize,right=9mm of o1] (otext) {Cycle A + Chain B + Cycle C};
        \node[font=\scriptsize,right=9mm of t1] (ttext) {Cycle A + Chain B$'$ + Cycle C};
        \draw[decorate,decoration={brace,amplitude=4pt},keporange] ($(otext.south west)+(16mm,-1mm)$) -- ($(ttext.north west)+(16mm,1mm)$)
          node[midway,right=3mm,font=\scriptsize,keporange] {small replacement};
      \end{tikzpicture}
    \end{column}

    \begin{column}{0.45\textwidth}
      \textbf{Simple implication}
      \[
        \gap(\ytwo) \le \epsilon
      \]
      \[
        \Longrightarrow
        \text{max possible improvement} \le \epsilon .
      \]

      \vspace{0.5em}
      \begin{alertblock}{Interpretation}
        \centering
        DFL can only remove regret that is actually present.
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Look Beyond Rank 1}
  \small
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \begin{block}{Diagnostic idea}
        If KEP has close substitutes, do not only inspect the selected solution.

        \vspace{0.35em}
        For each trained model and graph:
        \begin{enumerate}\tightitems
          \item solve for the best predicted KEP solution;
          \item exclude it with a no-good cut;
          \item re-solve for rank 2;
          \item evaluate both under true reward.
        \end{enumerate}
      \end{block}

      \begin{alertblock}{Question}
        \centering
        Does solution quality collapse after rank 1, or are alternatives still near-oracle?
      \end{alertblock}
    \end{column}

    \begin{column}{0.48\textwidth}
      \centering
      \begin{tikzpicture}[x=1mm,y=1mm]

\section{Takeaways}

\begin{frame}{Takeaways from this Project}
  \footnotesize
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.31\textwidth}
      \begin{block}{1. Decision, not prediction}
        We are not only fitting edge rewards. We study how prediction errors change downstream exchange decisions.
      \end{block}
    \end{column}

    \begin{column}{0.31\textwidth}
      \begin{exampleblock}{2. Real KEP observability}
        Real deployments cannot use retrospective oracle labels. We must separate topology, post-training ambiguity, and after-the-fact mechanism evidence.
      \end{exampleblock}
    \end{column}

    \begin{column}{0.31\textwidth}
      \begin{alertblock}{3. DFL is conditional}
        DFL can improve KEP decisions, but its effect is limited when two-stage already selects close substitutes.
      \end{alertblock}
    \end{column}
  \end{columns}

  \vspace{0.9em}
  \centering
  \begin{tikzpicture}
    \node[kcardwide,text width=0.88\textwidth,draw=keporange!75,fill=keporange!8] {
      \textbf{Conclusion:} DFL for KEP should be mechanism-aware: useful when it promotes near-oracle candidates, limited when close substitutes already exist, and risky when it promotes the wrong candidate.
    };
  \end{tikzpicture}

  \vspace{0.6em}
  {\scriptsize Next step: turn this mechanism story into a conference paper, not only a benchmark comparison.}
\end{frame}
